3. Let, Const, Var
i. Let- access with in the scope, changeable
ii. Const- no changeable
iii. Var- changeable, accessable any where, Global variable, Function-scoped (not block-scoped), Can be redeclared and updated

function test() {

    // VAR
    if (true) {
        var a = 10;
    }
    console.log("var a:", a); // ✅ 10 (not block scoped)

    // LET
    if (true) {
        let b = 20;
        console.log("let b inside:", b); // ✅ 20
    }
    // console.log("let b outside:", b); ❌ ERROR (block scoped)

    // CONST
    const c = 30;
    // c = 40; ❌ ERROR (cannot reassign)

    console.log("const c:", c); // ✅ 30
}
